#include <string>

using namespace std;

string solution(int n, int t, int m, vector<string> timetable) {
    string answer = "";
    int tmp_ans = 0;
    vector<int> table;
    for(int i=0;i<timetable.size();i++)
    {
        string word[2];
        int idx = 0;
        string temp = "";
        for(int j=0;j<timetable[i].size();j++)
        {
            if(timetable[i][j] == ':')
            {
                word[idx++] = temp;
                temp = "";
            }
            else temp += timetable[i][j];
        }
        word[idx++] = temp;
        int sum = stoi(word[0]) * 60 + stoi(word[1]);
        table.push_back(sum);
    }
    sort(table.begin(),table.end());
    int time = 540;
    while(n!=0)
    {
        int m2 = m;
        int last = 0;
        for(int i=0;i<table.size();i++)
        {
            if(table[i] <= time)
            {
                last = table[i];
                table.erase(table.begin()+i);
                m2--;
                i--;
            }
            if(m2==0) break;
        }
        if(n==1)
        {
            if(m2>0) tmp_ans = time;
            else if(m2==0) tmp_ans = last-1;
        }
        n--;
        time+=t;
    }
    string hour = to_string(tmp_ans/60);
    string min = to_string(tmp_ans%60);
    if(hour.size()==1) hour = '0'+hour;
    if(min.size()==1) min = '0'+min;
    answer = hour+':'+min;
    return answer;
}
