#include <string>
#include <vector>
#include <queue>
#include <algorithm>
#include <map>
using namespace std;
vector<string> go(string a,map<string,int> m)
{
    string save = a;
    vector<string> result;
    for(int i=0;i<a.size();i++)
    {
        a = save;
        int how = a[i] - 'a';
        for(int j=0;j<26;j++)
        {
            a[i] = 'a' + (how+j)%26;
            if(m[a]>0) result.push_back(a); 
        }
    }
    return result;
}
int solution(string begin, string target, vector<string> words) {
    int answer = 0;
    map<string,int> m;
    m[begin]++;
    for(int i=0;i<words.size();i++)
    {
        if(words[i] == begin) continue; //
        m[words[i]]++;
    }
    queue<string> q;
    q.push(begin);
    while(!q.empty())
    {
        string x = q.front();
        q.pop();
        vector<string> temp = go(x,m);
        for(int i=0;i<temp.size();i++)
        {
            if(m[temp[i]]==1 && temp[i] != x)
            {
                q.push(temp[i]);
                m[temp[i]] = m[x]+1;
            }
        }
    }
    if(m[target] == 1 || m[target] == 0) return 0;
    answer = m[target]-1;
    return answer;
}

#include <string>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
int check[60];
int solution(string begin, string target, vector<string> words) {
    int answer = 0;
    queue<string> q;
    q.push(begin);
    while(!q.empty())
    {
        string x = q.front();
        int idx = find(words.begin(),words.end(),x)-words.begin();
        if(find(words.begin(),words.end(),x)==words.end()) idx = 0;
        q.pop();
        for(int i=0;i<words.size();i++)
        {
            if(check[i]!=0) continue;
            int count = 0;
            for(int j=0;j<words[i].size();j++)
            {
                if(x[j]!=words[i][j]) count++;
            }
            if(count==1)
            {
                q.push(words[i]);
                check[i] = check[idx]+1;
            }
        }
    }
    int idx = -1;
    for(int i=0;i<words.size();i++)
    {
        if(words[i] == target)
        {
            idx = i;
            break;
        }
    }
    if(idx==-1) return 0;
    answer = check[idx];
    return answer;
}
