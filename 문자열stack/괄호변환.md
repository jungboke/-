#include <string>

bool checkArray(string a) // 재귀함수 어려움
{
    stack<char> s;
    bool check = true;
    for(int i=0;i<a.size();i++)
    {
        if(s.size() == 1 && s.top() == ')')
        {
            break;
        }
        if(s.empty() || (s.top() == a[i]))
        {
            s.push(a[i]);
        }
        else
            s.pop();
    }
    if(!s.empty()) check = false;
    return check;
}
string fix(string a)
{
    if(a.size() == 0) return a;
    string u = "";
    string v = "";
    int count1 = 0;
    int count2 = 0;
    for(int i=0;i<a.size();i++)
    {
        if(a[i] == '(') count1++;
        else count2++;
        
        if(count2 == count1)
        {
            u = a.substr(0,i+1);
            v = a.substr(i+1);
            break;
        }    
    }
    if(checkArray(u) == true) return u + fix(v);
    else
    {
        string temp2 = "";
        temp2 += '(';
        temp2 += fix(v);
        temp2 += ')';
        u = u.substr(1, u.size()-2);
        for(int i=0;i<u.size();i++)
        {
            if(u[i] == '(') temp2 += ')';
            else temp2 += '(';
        }
        return temp2;
    }
}
string solution(string p) {
    string answer = "";
    if(checkArray(p) == true) return p;
    return fix(p);
}

#include <string>
#include <vector>
#include <stack>
using namespace std;
bool check(string p)
{
    stack<char> st;
    for(int i=0;i<p.size();i++)
    {
        if(st.size() == 1 && st.top() == ')') return false;
        if(st.empty() || (st.top() == p[i])) st.push(p[i]);
        else st.pop();
    }
    if(st.size() != 0) return false;
    else return true;
}
string transform(string p)
{
    if(p.size() == 0) return p;
    int left = 0;
    int right = 0;
    string u = "";
    string v = "";
    for(int i=0;i<p.size();i++)
    {
        if(p[i] == '(') left++;
        else  right++;
        
        if(left == right)
        { 
            u = p.substr(0,i+1);
            v = p.substr(i+1,p.size()-i+1);
            if(check(u) == true) return u+transform(v);
            else
            {
                string temp = "";
                temp = temp + '(' + transform(v) + ')';
                u.erase(u.begin());
                u.erase(u.end()-1);
                for(int i=0;i<u.size();i++)
                {
                    if(u[i] == '(') temp += ')';
                    else temp += '(';
                }
                return temp;
            }
        }
    }
}
string solution(string p) {
    string answer = "";
    if(check(p) == true)
    {
        answer = p;
        return answer;
    }
    answer = transform(p);
    return answer;
}

#include <string>
#include <vector>
#include <stack>
using namespace std;
bool correct(string p)
{
    bool flag = true;
    stack<char> st;
    for(int i=0;i<p.size();i++)
    {
        if(st.empty() || st.top()==p[i]) st.push(p[i]);
        else st.pop();
        
        if(!st.empty() && st.top() == ')')
        {
            flag = false;
            return flag;
        }
    }
    if(st.size()!=0) flag = false;
    return flag;
}
string go(string p)
{
    if(p.size()==0) return p;
    int left = 0;
    int right = 0;
    string u = "";
    string v = "";
    for(int i=0;i<p.size();i++)
    {
        if(p[i] == ')') right++;
        else if(p[i] == '(') left++;
        
        if(left==right)
        {
            u = p.substr(0,i+1);
            v = p.substr(i+1,p.size()-u.size());
            break;
        }
    }
    if(correct(u)==true) return u+go(v);
    else
    {
        string temp = "";
        temp = temp + '(' + go(v) + ')';
        u.erase(u.begin());
        u.erase(u.end()-1);
        for(int i=0;i<u.size();i++)
        {
            if(u[i] == '(') u[i] = ')';
            else if(u[i] == ')') u[i] = '(';
        }
        temp += u;
        return temp;
    }
}
string solution(string p) {
    string answer = "";
    if(correct(p) == true) return p;
    answer = go(p);
    return answer;
}
