#include <string>
#include <iostream>
#include <stack>
using namespace std;

bool solution(string s)
{
    bool answer = true;
    stack<char> st;
    for(int i=0;i<s.size();i++)
    {
        if(st.empty() || st.top() == s[i]) st.push(s[i]);
        else st.pop();
        if(!st.empty() && st.top() == ')')
        {
            answer = false;
            break;
        }
    }
    if(st.size() != 0) answer = false;
    return answer;
}
