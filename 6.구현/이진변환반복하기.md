#include <string>
#include <vector>

using namespace std;

vector<int> solution(string s) {
    vector<int> answer;
    int zero = 0;
    int count = 0;
    while(s!="1")
    {
        count++;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='0')
            {
                zero++;
                s.erase(s.begin()+i);
                i--;
            }
        }
        int num = s.size();
        string temp = "";
        while(num!=0)
        {
            temp = to_string(num%2) + temp;
            num = num / 2;
        }
        s = temp;
    }
    answer.push_back(count);
    answer.push_back(zero);
    return answer;
}
