#include <string>
#include <vector>
#include <queue>
using namespace std;
vector<int> a[20001];
int dist[20001];
int solution(int n, vector<vector<int>> edge) {
    int answer = 0;
    for(int i=0;i<edge.size();i++)
    {
        a[edge[i][0]].push_back(edge[i][1]);
        a[edge[i][1]].push_back(edge[i][0]);
    }
    queue<int> q;
    q.push(1);
    dist[1] = 1;
    while(!q.empty())
    {
        int x = q.front();
        q.pop();
        for(int k : a[x])
        {
            if(dist[k]==0)
            {
                q.push(k);
                dist[k] = dist[x] + 1;
            }
        }
    }
    int maxi = 0;
    for(int i=1;i<=n;i++)
    {
        if(dist[i]>maxi) maxi = dist[i];
    }
    for(int i=1;i<=n;i++)
    {
        if(dist[i]==maxi) answer++;
    }
    return answer;
}
