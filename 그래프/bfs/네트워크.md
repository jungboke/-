#include <string>
#include <vector>
#include <queue>
using namespace std;
void dfs(int num,vector<bool> &check,vector<vector<int>> &a)
{
    check[num] = true;
    for(int k : a[num])
    {
        if(check[k] == false) dfs(k,check,a);
    }
}
int solution(int n, vector<vector<int>> computers) {
    int answer = 0;
    vector<vector<int>> a(n);
    vector<bool> check(n);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(i==j) continue;
            if(computers[i][j] == 1) a[i].push_back(j);
        }
    }
    for(int i=0;i<n;i++)
    {
        if(check[i] == false)
        {
            dfs(i,check,a);
            answer++;
        }
    }
    return answer;
}

#include <string>
#include <vector>
#include <queue>
using namespace std;
bool check[300];
int solution(int n, vector<vector<int>> computers) {
    int answer = 0;
    for(int i=0;i<computers.size();i++)
    {
        if(check[i] == true) continue;
        answer++;
        queue<int> q;
        q.push(i);
        check[i] = true;
        while(!q.empty())
        {
            int x = q.front();
            q.pop();
            for(int i=0;i<computers[x].size();i++)
            {
                if(i==x) continue;
                if(computers[x][i] == 1 && check[i] == false)
                {
                    q.push(i);
                    check[i] = true;
                }
            }
        }
    }
    return answer;
}

#include <string>
#include <vector>

using namespace std;
vector<int> a[201];
bool check[201];
void dfs(int num)
{
    check[num] = true;
    for(int i=0;i<a[num].size();i++)
    {
        int k = a[num][i];
        if(check[k]==false)
        {
            dfs(k);
        }
    }
}
int solution(int n, vector<vector<int>> computers) {
    int answer = 0;
    for(int i=0;i<computers.size();i++)
    {
        for(int j=0;j<computers[i].size();j++)
        {
            if(i==j) continue;
            if(computers[i][j]==1)
            {
                a[i+1].push_back(j+1);
            }
        }
    }
    for(int i=1;i<=n;i++)
    {
        if(check[i]==false)
        {
            answer++;
            dfs(i);
        }
    }
    return answer;
}
