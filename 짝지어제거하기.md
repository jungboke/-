#include <iostream>
#include<string>
#include<stack>
using namespace std;

int solution(string s) // stack 활용법 중복되는 문자 제거
{
    int answer = 0;
    stack<int> temp;
    for(int i=0;i<s.size();i++)
    {
        if(temp.empty() || temp.top() != s[i]) temp.push(s[i]);
        else temp.pop();
    }
    if(temp.size() > 0) answer = 0;
    else answer = 1;
    return answer;
}
