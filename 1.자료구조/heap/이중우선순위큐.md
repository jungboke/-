```
#include <string>
#include <vector>
#include <queue>
using namespace std;

vector<int> solution(vector<string> operations) {
    vector<int> answer;
    priority_queue<int,vector<int>,greater<int>> minq;
    priority_queue<int> maxq;
    int count = 0;
    for(int i=0;i<operations.size();i++)
    {
        if(count==0)
        {
            while(!minq.empty()) minq.pop();
            while(!maxq.empty()) maxq.pop();
        }
        int x = stoi(operations[i].substr(2,operations[i].size()-2));
        if(operations[i][0] == 'I')
        {
            minq.push(x);
            maxq.push(x);
            count++;
        }
        else if(count == 0) continue;
        else if(x == -1)
        {
            minq.pop();
            count--;
        }
        else if(x == 1)
        {
            maxq.pop();
            count--;
        }
    }
    if(count == 0)
    {
        answer.push_back(0);
        answer.push_back(0);
    }
    else
    {
        answer.push_back(maxq.top());
        answer.push_back(minq.top());
    }
    return answer;
}
```

```
#include <string>
#include <vector>
#include <set>
using namespace std;

vector<int> solution(vector<string> operations) {
    vector<int> answer;
    multiset<int> ms;
    for(int i=0;i<operations.size();i++)
    {
        if(operations[i][0]=='I')
        {
            int num = stoi(operations[i].substr(2));
            ms.insert(num);
        }
        else if(operations[i][0]=='D')
        {
            if(ms.empty()) continue;
            int num = stoi(operations[i].substr(2));
            if(num==1)
            {
                auto it = ms.end();
                it--;
                ms.erase(it);
            }
            else if(num==-1)
            {
                ms.erase(ms.begin());
            }
        }
    }
    if(ms.empty())
    {
        answer.push_back(0);
        answer.push_back(0);
    }
    else
    {
        auto it = ms.end();
        it--;
        answer.push_back(*it);
        answer.push_back(*ms.begin());
    }
    return answer;
}
```
+ minheap과 maxheap 두개를 만들어서 가상이중우선순위큐를 제작.(틀림)
+ list나 배열중 자동sort되는 set에서 중복숫자 가능한 multiset을 사용하여 우선순위큐 대용으로 사용.
+ substr()에서 두번째 인자가 들어가지않으면 문자열의 마지막까지, list에서는 iterator 연산(--,++)만 가능.
