#include <string>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

int solution(string name) {
    int answer = 0;
    string origin = "";
    for(int i=0;i<name.size();i++)
    {
        origin += 'A';
    }
    int k = name.size();
    int now = 0;
    while(k--)
    {
        if(name == origin) break;
        vector<pair<int,int>> result;
        for(int i=0;i<name.size();i++)
        {
            if(name[i] != origin[i])
            {
                int m = abs(i-now);
                int m2 = name.size();
                int dis = min(m,m2-m);
                result.push_back(make_pair(dis,i));
            }
        }
        auto it = min_element(result.begin(),result.end());
        answer += it->first;
        now = it->second;
        answer+= min(name[now]-'A',26-(name[now]-'A'));
        origin[now] = name[now];
    }
    return answer;
}
