#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int solution(vector<int> people, int limit) {
    int answer = 0;
    int first = 0;
    sort(people.begin(),people.end());
    for(int i=people.size()-1;i>=first;i--)
    {
        if(i==first)
        {
            answer++;
            break;
        }
        if(people[first] + people[i] <= limit)
        {
            answer++;
            first++;
        }
        else answer++;
    }
    return answer;
}
