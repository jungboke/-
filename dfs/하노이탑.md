#include <string>
#include <vector>

using namespace std;
vector<vector<int>> result;
void go(int a,int now,int next)
{
    if(a==0) return;
    go(a-1,now,6-(now+next));
    vector<int> temp;
    temp.push_back(now);
    temp.push_back(next);
    result.push_back(temp);
    go(a-1,6-(now+next),next);
}
vector<vector<int>> solution(int n) {
    vector<vector<int>> answer;
    go(n,1,3);
    answer = result;
    return answer;
}
