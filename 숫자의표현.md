#include <string>
#include <vector>

using namespace std;

int solution(int n) {
    int answer = 0;
    for(int i=1;i<=n;i++)
    {
        int num = 0;
        int k = i;
        while(num <= n)
        {
            num += k;
            if(num == n)
            {
                answer++;
                break;
            }
            k++;
        }
    }
    return answer;
}
