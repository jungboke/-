#include <string>
#include <vector>
#include <queue>
using namespace std;

int solution(vector<int> priorities, int location) {
    int answer = 0;
    int count = 0;
    queue<pair<int,int>> q;
    for(int i=0;i<priorities.size();i++)
    {
        q.push(make_pair(priorities[i],i));
        
    }
    while(!q.empty())
    {
        auto x = q.front();
        for(int i=0;i<priorities.size();i++)
        {
            if(priorities[i]>x.first)
            {
                q.pop();
                q.push(x);
                break;
            }
            if(i==priorities.size()-1)
            {
                q.pop();
                count++;
                priorities.erase(priorities.begin() + x.second);
                //priorities[x.second] = 0;
                if(x.second==location)
                {
                    answer = count;
                    return answer;
                }
            }
        }
    }
}

#include <string>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

int solution(vector<int> priorities, int location) {
    int answer = 0;
    queue<pair<int,int>> q;
    for(int i=0;i<priorities.size();i++)
        q.push(make_pair(priorities[i],i));
    while(!q.empty())
    {
        auto x = q.front();
        q.pop();
        if(x.first < *max_element(priorities.begin(),priorities.end()))
            q.push(x);
        else
        {
            answer++;
            priorities.erase(max_element(priorities.begin(),priorities.end()));
            if(x.second == location) break;
        }
    }
    return answer;
}
