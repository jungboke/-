#include <string>
#include <vector>
#include <algorithm>
#include <set>
using namespace std;
bool check[10000];
vector<string> answer;
void dfs(string start,int count,vector<string>temp,vector<vector<string>> tickets)
{
    temp.push_back(start);
    if(count == tickets.size())
    {
        if(answer.size()==0) answer = temp;
        return;
    }
    for(int i=0;i<tickets.size();i++)
    {
        if(tickets[i][0] == start && check[i] == false)
        {
            check[i] = true;
            dfs(tickets[i][1],count+1,temp,tickets);
            check[i] = false;
        }
    }
    temp.pop_back();
}
vector<string> solution(vector<vector<string>> tickets) {
    vector<string> temp;
    sort(tickets.begin(),tickets.end());
    dfs("ICN",0,temp,tickets);
    return answer;
}

#include <string>
#include <vector>
#include <algorithm>
using namespace std;
bool check[10001];
vector<string> result;
bool cmp(vector<string> a,vector<string> b)
{
    return (a[1]<b[1]);
}
void dfs(string a,vector<vector<string>> tickets,int cnt,vector<string> vec)
{
    if(cnt==tickets.size())
    {
        vec.push_back(a);
        if(result.size()==0) result = vec;
        return;
    }
    for(int i=0;i<tickets.size();i++)
    {
        if(check[i]==true) continue;
        if(tickets[i][0] == a)
        {
            check[i] = true;
            vec.push_back(tickets[i][0]);
            dfs(tickets[i][1],tickets,cnt+1,vec);
            vec.pop_back();
            check[i] = false;
        }
    }
}
vector<string> solution(vector<vector<string>> tickets) {
    vector<string> answer;
    sort(tickets.begin(),tickets.end(),cmp);
    vector<string> vec;
    dfs("ICN",tickets,0,vec);
    answer = result;
    return answer;
}
