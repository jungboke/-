int dy[] = {0,1,0,-1};

    int number_of_area = 0;
    int max_size_of_one_area = 0;
    
    vector<int> answer;
    queue<pair<int,int>> q;
    vector<int> result;
    int many = 0;
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<n;j++)
        {
            check[i][j] = false;
        }
    }
    
    
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(picture[i][j] != 0 && check[i][j] == false)
            {
                int color = picture[i][j];
                int count = 0;
                q.push(make_pair(i,j));
                check[i][j] = true;
                while(!q.empty())
                {
                    int x = q.front().first;
                    int y = q.front().second;
                    q.pop();
                    count++;
                    for(int i=0;i<4;i++)
                    {
                        int nx = x + dx[i];
                        int ny = y + dy[i];
                        if(check[nx][ny] == false &&nx>=0&&nx<m&&ny>=0&&ny<n)
                        {
                            if(picture[nx][ny] == color)
                            {
                                q.push(make_pair(nx,ny));
                                check[nx][ny] = true;
                            }
                        }
                    }
                }
                result.push_back(count);
                many++;
            }
        }
    }
    int max = *max_element(result.begin(),result.end());
    answer.push_back(many);
    answer.push_back(max);
    return answer;
}

#include <vector>
#include <queue>
#include <algorithm>
using namespace std;
int dx[] = {-1,0,1,0};
int dy[] = {0,1,0,-1};
bool check[100][100];
// 전역 변수를 정의할 경우 함수 내에 초기화 코드를 꼭 작성해주세요.

vector<int> solution(int m, int n, vector<vector<int>> picture) {
    vector<int> answer(2);
    int many = 0;
    int how = 0;
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<n;j++) check[i][j] = false;
    }
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(picture[i][j] != 0 && check[i][j] == false)
            {
                int count = 0;
                many++;
                queue<pair<int,int>> q;
                q.push(make_pair(i,j));
                check[i][j] = true;
                while(!q.empty())
                {
                    int x,y;
                    tie(x,y) = q.front();
                    q.pop();
                    count++;
                    for(int i=0;i<4;i++)
                    {
                        int nx = x+dx[i];
                        int ny = y+dy[i];
                        if(nx>=0&&nx<m&&ny>=0&&ny<n&&check[nx][ny] == false)
                        {
                            if(picture[nx][ny] == picture[x][y])
                            {
                                q.push(make_pair(nx,ny));
                                check[nx][ny] = true;
                            }
                        }
                    }
                }
                if(how<count) how = count;
            }
        }
    }
    answer[0] = many;
    answer[1] = how;
    return answer;
}
