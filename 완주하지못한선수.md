1번째 답
```
#include <string>
#include <vector>
#include <unordered_map>
using namespace std;

string solution(vector<string> participant, vector<string> completion) {
    string answer = "";
    unordered_map<string,int> m;
    for(int i=0;i<participant.size();i++)
    {
        m[participant[i]]++;
    }

    for(int i=0;i<completion.size();i++)
    {
        m[completion[i]]--;
    }
    for(auto pair : m)
    {
        if(pair.second != 0)
        {
            answer = pair.first;
        }
    }
    return answer;
}
```

2번째 답
```
#include <string>
#include <vector>
#include <unordered_map>
using namespace std;

string solution(vector<string> participant, vector<string> completion) {
    string answer = "";
    unordered_map<string,int> m;
    for(int i=0;i<participant.size();i++)
    {
        m[participant[i]]++;
    }
    for(int i=0;i<completion.size();i++)
    {
        m[completion[i]]--;
    }
    unordered_map<string,int>::iterator it;
    for(it=m.begin();it!=m.end();it++)
    {
        if(it->second == 1)
        {
            answer = it->first;
            break;
        }
    }
    return answer;
}
```
