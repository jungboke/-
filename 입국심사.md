#include <string>
#include <vector>
#include <algorithm>
using namespace std;

long long solution(int n, vector<int> times) {
    long long answer = 0;
    bool check[times.size()];
    sort(times.begin(),times.end());
    vector<int> tmp_times = times;
    while(n!=0)
    {
        for(int i=0;i<times.size();i++)
        {
            if(check[i] == false)
            {
                check[i] = true;
                n--;
            }
        }
        for(int i=0;i<times.size();i++)
        {
            if(check[i] == true)
            {
                times[i]--;
            }
        }
        for(int i=0;i<times.size();i++)
        {
            if(times[i] == 0)
            {
                times[i] = tmp_times[i];
                check[i] = false;
            }
        }
        answer++;
    }
    answer--;
    return answer;
}

#include <string>
#include <vector>
#include <algorithm>
using namespace std;

long long solution(int n, vector<int> times) {
    long long answer = 0;
    long long start = 1;
    long long how = *max_element(times.begin(),times.end());
    long long end = how * n;
    answer = end;
    while(start<=end)
    {
        long long mid = (start+end) / 2;
        long long sum = 0;
        for(int i=0;i<times.size();i++) sum += mid/times[i];
        if(sum < n) start = mid + 1;
        else
        {
            if(mid<answer) answer = mid;
            end = mid - 1;
        }
    }
    return answer;
}
