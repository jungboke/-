#include <string>
#include <vector>
#include <algorithm>
using namespace std;

string solution(string s) {
    string answer = "";
    vector<int> result;
    bool flag = false;
    int temp = 0;
    s = s + ' ';
    for(int i=0;i<s.size();i++)
    {
        if(s[i] == ' ')
        {
            if(flag==true) result.push_back(0-temp);
            else result.push_back(temp);
            temp = 0;
            flag = false;
            continue;
        }
        if(s[i] == '-')
        {
            flag = true;
            continue;
        }
        temp = temp*10 + (s[i]-48);
    }
    
    answer = to_string(*min_element(result.begin(),result.end()));
    answer += ' ';
    answer += to_string(*max_element(result.begin(),result.end()));
    return answer;
}
