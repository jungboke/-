#include <vector>

using namespace std;

vector<int> solution(vector<string> gems) {
    vector<int> answer;
    set<string> s(gems.begin(),gems.end());
    int num = s.size();
    unordered_map<string,int> m;
    int start = 0;
    int end = 0;
    int result[2];
    for(int i=0;i<gems.size();i++)
    {
        m[gems[i]]++;
        if(m.size() == num)
        {
            end = i;
            break;
        }
    }
    result[0] = start+1;
    result[1] = end+1;
    int min = end - start + 1;
    while(end<gems.size())
    {
        string key = gems[start];
        m[key]--;
        start++;
        if(m[key] == 0)
        {
            end++;
            for(;end<gems.size();end++)
            {
                m[gems[end]]++;
                if(gems[end] == key)
                {
                    break;
                }
            }
            if(end == gems.size()) break;
        }
        if(end-start+1 < min)
        {
            result[0] = start+1;
            result[1] = end+1;
            min = end-start+1;
        }
    }
    answer.push_back(result[0]);
    answer.push_back(result[1]);
    return answer;
}
