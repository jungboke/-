// bfs문제
// 두 P사이의 거리(X면 돌아감)이 2이상이 나오면 0, 아니면 1
// 완탐으로 모든 P마다 bfs시전
// 매번 check,dist 배열 초기화해주고 bfs후 P가 들어있는 배열의 dist값이 2이상이면 0
#include <string>
#include <vector>
#include <algorithm>
#include <string.h>
#include <queue>
#include <tuple>
using namespace std;
int dx[] = {0,-1,0,1};
int dy[] = {-1,0,1,0};
int dist[10][10];
int bfs(vector<string> a,int x,int y)
{
    memset(dist,-1,sizeof(dist));
    queue<pair<int,int>> q;
    q.push(make_pair(x,y));
    dist[x][y] = 0;
    while(!q.empty())
    {
        int x,y;
        tie(x,y) = q.front();
        q.pop();
        for(int i=0;i<4;i++)
        {
            int nx = x+dx[i];
            int ny = y+dy[i];
            if(nx>=0&&nx<5&&ny>=0&&ny<5&&dist[nx][ny]==-1)
            {
                if(a[nx][ny]!='X')
                {
                    q.push(make_pair(nx,ny));
                    dist[nx][ny] = dist[x][y]+1;
                }
            }
        }
    }
    for(int i=0;i<5;i++)
    {
        for(int j=0;j<5;j++)
        {
            if(i==x&&j==y) continue;
            if(a[i][j]=='P'&&dist[i][j]<=2&&dist[i][j]!=-1)
            {
                return -1;
            }
        }
    }
    return 1;
}

vector<int> solution(vector<vector<string>> places) {
    vector<int> answer;
    for(int c=0;c<5;c++)
    {
        vector<string> temp = places[c];
        bool flag = true;
        for(int i=0;i<5;i++)
        {
            for(int j=0;j<5;j++)
            {
                if(temp[i][j]=='P')
                {
                    if(bfs(temp,i,j)==-1)
                    {
                        flag = false;
                        break;
                    }
                }
            }
            if(flag==false) break;
        }
        if(flag==false) answer.push_back(0);
        else answer.push_back(1);
    }
    return answer;
}
