#include <string>
#include <vector>
#include <queue>
using namespace std;
void dfs(int num,vector<bool> &check,vector<vector<int>> &a)
{
    check[num] = true;
    for(int k : a[num])
    {
        if(check[k] == false) dfs(k,check,a);
    }
}
int solution(int n, vector<vector<int>> computers) {
    int answer = 0;
    vector<vector<int>> a(n);
    vector<bool> check(n);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(i==j) continue;
            if(computers[i][j] == 1) a[i].push_back(j);
        }
    }
    for(int i=0;i<n;i++)
    {
        if(check[i] == false)
        {
            dfs(i,check,a);
            answer++;
        }
    }
    return answer;
}
