#include <string>
#include <vector>
#include <algorithm>
using namespace std;

string solution(int n, int t, int m, vector<string> timetable) {
    string answer = "";
    int answer2 = 0;
    vector<int> result;
    for(int i=0;i<timetable.size();i++)
    {
        timetable[i] += ':';
        int time[2];
        int idx = 0;
        string temp = "";
        for(int j=0;j<timetable[i].size();j++)
        {
            if(timetable[i][j] == ':')
            {
                time[idx++] = stoi(temp);
                temp = "";
            }
            else temp += timetable[i][j];
        }
        result.push_back(time[0]*60+time[1]);
    }
    sort(result.begin(),result.end());
    int time = 540;
    int end = 0;
    int m2 = m;
    while(n--)
    {  
        m = m2;
        for(int i=0;i<result.size();i++)
        {
            if(m==0) break;
            if(result[i] <= time)
            {
                end = result[i];
                result.erase(result.begin()+i);
                m--;
                i--;
            }
        }
        if(n==0)
        {
            if(m>0) answer2 = time;
            else if(m==0) answer2 = end-1;
        }
        time += t;
    }
    string hour = to_string(answer2/60);
    if(hour.size() == 1) hour = '0' + hour;
    string min = to_string(answer2%60);
    if(min.size() == 1) min = '0' + min;
    answer = hour + ':' + min;
    return answer;
}
