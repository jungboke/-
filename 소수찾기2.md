#include <string>
#include <vector>

using namespace std;

int solution(int n) {
    int answer = 0;
    vector<bool> check(n+1,0); // 0 소수 1 소수 x
    check[0] = 1;
    check[1] = 1;
    for(int i=2;i<=n;i++)
    {
        if(check[i]) continue;
        for(int j=i+i;j<=n;j+=i)
        {
            check[j] = 1;
        }
    }
    for(int i=1;i<=n;i++)
    {
        if(check[i] == 0) answer++;
    }
    return answer;
}
