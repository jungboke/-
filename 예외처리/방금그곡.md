#include <string>
#include <vector>
#include <algorithm>
using namespace std;

string solution(string m, vector<string> musicinfos) {
    string answer = "";
    int max = 0;
    for(int i=0;i<musicinfos.size();i++)
    {
        string token[6];
        int idx = 0;
        string temp = "";
        for(int j=0;j<musicinfos[i].size();j++)
        {
            if(musicinfos[i][j] == ',' || musicinfos[i][j] == ':')
            {
                token[idx++] = temp;
                temp = "";
            }
            else temp += musicinfos[i][j];
        }
        token[idx] = temp;
        temp = "";
        int hour = stoi(token[2]) - stoi(token[0]);
        int minute = stoi(token[3]) - stoi(token[1]);
        int length = hour * 60 + minute;
        for(int i=0;i<length;i++)
        {
            temp += token[5][i%(token[5].size())];
            if(token[5][(i+1)%(token[5].size())] == '#') length++; //
        }
        for(int i=0;i<temp.size();i++)
        {
            string test = temp.substr(i,m.size()); //
            if(test == m)
            {
                if(temp[i+m.size()] != '#')
                    if(length>max)
                    {
                        answer = token[4];
                        max = length;
                    }
            }
        }
    }
    if(answer == "") return "(None)";
    return answer;
}

#include <string>
#include <vector>
#include <map>
using namespace std;
string change(string m,map<string,char> map1)
{
    string result = "";
    for(int i=0;i<m.size();i++)
    {
        string temp = m.substr(i,2);
        if(map1[temp]!=0)
        {
            result += map1[temp];
            i++;
        }
        else result += m[i];
    }
    return result;
}
vector<string> go(string a,string m)
{
    vector<string> vec;
    string token[6];
    int idx = 0;
    string temp = "";
    for(int j=0;j<a.size();j++)
    {
        if(a[j]==':'||a[j]==',')
        {
            token[idx++] = temp;
            temp = "";
        }
        else temp+=a[j];
    }
    token[idx++] = temp;
    int time = (stoi(token[2])-stoi(token[0]))*60+(stoi(token[3])-stoi(token[1]));
    vec.push_back(to_string(time));
    vec.push_back(token[4]);
    vec.push_back(token[5]);
    return vec;
}
string solution(string m, vector<string> musicinfos) {
    string answer = "";
    map<string,char> map1;
    map1["A#"] = 'H';
    map1["C#"] = 'I';
    map1["D#"] = 'J';
    map1["F#"] = 'K';
    map1["G#"] = 'L';
    m = change(m,map1);
    int maxi = 0;
    for(int i=0;i<musicinfos.size();i++)
    {
        vector<string> k = go(musicinfos[i],m);
        int time = stoi(k[0]);
        k[2] = change(k[2],map1);
        string word = "";
        for(int j=0;j<time;j++)
        {
            word+=k[2][j%k[2].size()];
        }
        if(word.find(m)!=string::npos)
        {
            if(maxi<stoi(k[0]))
            {
                maxi = stoi(k[0]);
                answer = k[1];
            }
        }
    }
    if(answer=="") answer = "(None)";
    return answer;
}
