#include <string>
#include <vector>
#include <set>
#include <algorithm>
using namespace std;

vector<int> solution(int n, int s) {
    vector<int> answer;
    for(int i=0;i<n;i++)
    {
        answer.push_back(s/n);
    }
    int tmp_s = s%n;
    for(int i=0;i<tmp_s;i++)
    {
        answer[i]++;
    }
    if((s/n) == 0)
    {
        answer.clear();
        answer.push_back(-1);
    }
    sort(answer.begin(),answer.end());
    return answer;
}
