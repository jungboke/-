#include <string>
#include <vector>
#include <stack>
using namespace std;
bool check(string a)
{
    stack<char> st;
    for(int i=0;i<a.size();i++)
    {
        if(st.empty() || a[i] == '[' || a[i] == '{' || a[i] == '(') st.push(a[i]);
        else
        {
            if(st.top()=='[')
            {
                if(a[i]==']') st.pop();
                else st.push(a[i]);
            }
            else if(st.top()=='{')
            {
                if(a[i]=='}') st.pop();
                else st.push(a[i]);
            }
            else if(st.top()=='(')
            {
                if(a[i]==')') st.pop();
                else st.push(a[i]);
            }
        }
    }
    if(st.size() != 0) return false;
    else return true;
}
int solution(string s) {
    int answer = 0;
    for(int i=0;i<s.size();i++)
    {
        string temp = s.substr(0,1);
        s.erase(s.begin(),s.begin()+1);
        s = s + temp;
        if(check(s) == true) answer++;
    }
    return answer;
}

        if(st.empty()||s[i] == '('||s[i] == '['||s[i] == '{') st.push(s[i]);
        else
        {
            if(st.top()=='('&&s[i]!=')') return;
            if(st.top()=='['&&s[i]!=']') return;
            if(st.top()=='{'&&s[i]!='}') return;
            st.pop();
        }
        if(!st.empty())
        {
            if(st.top()==')'||st.top()==']'||st.top()=='}') return;
        }
    }
    if(st.size() != 0) return;
    count++;
    return;
}
int solution(string s) {
    int answer = 0;
    string temp = "";
    for(int i=0;i<s.size();i++)
    {
        string s2 = s;
        temp = s2.substr(0,i);
        s2.erase(s2.begin(),s2.begin()+i);
        s2 += temp;
        check(s2);
    }
    answer = count;
    return answer;
}

#include <string>

bool check(string s)
{
    stack<char> st;
    bool flag = true;
    for(int i=0;i<s.size();i++)
    {
        if(st.empty()||s[i]=='['||s[i]=='('||s[i]=='{') st.push(s[i]);
        else
        {
            if(st.top()=='['&&s[i]==']') st.pop();
            else if(st.top()=='('&&s[i]==')') st.pop();
            else if(st.top()=='{'&&s[i]=='}') st.pop();
            else
            {
                flag = false;
                break;
            }
        }
        
        if(!st.empty())
        {
            if(st.top()==']'||st.top()==')'||st.top()=='}')
            {
                flag = false;
                break;
            }
        }
    }
    if(!st.empty()) flag = false;
    return flag;
}
int solution(string s) {
    int answer = 0;
    for(int i=0;i<s.size();i++)
    {
        s+=s.substr(0,1);
        s.erase(s.begin());
        if(check(s)==true) answer++;
    }
    return answer;
}
