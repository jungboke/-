#include <string>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

int solution(vector<int> scoville, int K) {
    int answer = 0;
    priority_queue<int,vector<int>,greater<int>> q;
    for(int i=0;i<scoville.size();i++)
    {
        q.push(scoville[i]);
    }
    int num = 0;
    while(q.top()<K)
    {
        if(q.size() == 1)
        {
            if(q.top() >= K) break;
            else return -1;
        }
        int x = q.top();
        q.pop();
        int y = q.top();
        q.pop();
        num = x + 2*y;
        q.push(num);
        answer++;
    }
    return answer;
}
