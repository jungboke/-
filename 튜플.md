    vector<int> answer;
    vector<int> sets;
    vector<pair<int,vector<int>>> sets2;
    set<int> result;
    int temp = 0;
    for(int i=0;i<s.size();i++)
    {
        if(s[i] == '{') continue;
        else if(s[i] == ',')
        {
            if(temp != 0)
            {
                sets.push_back(temp);
                temp = 0;
            }
            else continue;
        }
        else if(s[i] == '}')
        {
            sets.push_back(temp);
            sets2.push_back(make_pair(sets.size(),sets));
            sets.clear();
            temp = 0;
        }
        else
            temp = temp * 10 + s[i] - 48;
    }
    sets2.erase(sets2.end());
    sort(sets2.begin(),sets2.end());
    
    for(int i=0;i<sets2.size();i++)
    {
        for(int k:sets2[i].second)
        {
            if (result.find(k) == result.end())
            {
                result.insert(k);
                answer.push_back(k);
            }
        }
    }
    
    return answer;
}

#include <string>
#include <vector>
#include <cctype>
#include <algorithm>
using namespace std;
bool cmp(vector<int> a, vector<int> b)
{
    return (a.size() < b.size());
}
vector<int> solution(string s) {
    vector<int> answer;
    vector<vector<int>> result;
    vector<int> temp;
    int number = 0;
    for(int i=0;i<s.size()-1;i++)
    {
        if(isdigit(s[i]) != 0) number = number * 10 + (s[i]-48);
        else if(s[i] == ',' && number != 0)
        {
            temp.push_back(number);
            number = 0;
        }
        else if(s[i] == '}')
        {
            temp.push_back(number);
            number = 0;
            result.push_back(temp);
            temp.clear();
        }
    }
    sort(result.begin(),result.end(),cmp);
    vector<int> save;
    for(int i=0;i<result.size();i++)
    {
        for(int j=0;j<result[i].size();j++)
        {
            if(find(save.begin(),save.end(),result[i][j]) == save.end())
            {
                answer.push_back(result[i][j]);
                save.push_back(result[i][j]);
            }
        }
    }
    return answer;
}
