#include <string>
#include <vector>
#include <stack>
using namespace std;
bool check(string a)
{
    stack<char> st;
    for(int i=0;i<a.size();i++)
    {
        if(st.empty() || a[i] == '[' || a[i] == '{' || a[i] == '(') st.push(a[i]);
        else
        {
            if(st.top()=='[')
            {
                if(a[i]==']') st.pop();
                else st.push(a[i]);
            }
            else if(st.top()=='{')
            {
                if(a[i]=='}') st.pop();
                else st.push(a[i]);
            }
            else if(st.top()=='(')
            {
                if(a[i]==')') st.pop();
                else st.push(a[i]);
            }
        }
    }
    if(st.size() != 0) return false;
    else return true;
}
int solution(string s) {
    int answer = 0;
    for(int i=0;i<s.size();i++)
    {
        string temp = s.substr(0,1);
        s.erase(s.begin(),s.begin()+1);
        s = s + temp;
        if(check(s) == true) answer++;
    }
    return answer;
}
