#include <string>
#include <vector>
#include <unordered_map>
using namespace std;

vector<int> solution(string msg) {
    vector<int> answer;
    unordered_map<string,int> m;
    int idx = 1;
    for(int i=0;i<26;i++)
    {
        char ch = 'A' + i;
        string temp = "";
        temp += ch;
        m[temp] = idx++; 
    }
    for(int i=0;i<msg.size();i++)
    {
        string temp = "";
        string result = "";
        for(int j=i;j<msg.size();j++)
        {
            temp += msg[j];
            if(m.count(temp) != 0) result = temp;  
        }
        answer.push_back(m[result]);
        i = i + (result.size()-1);
        temp = result + msg[i+1];//
        m[temp] = idx++;
    }
    return answer;
}
