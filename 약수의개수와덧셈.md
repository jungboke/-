#include <string>
#include <vector>
#include <cmath>
using namespace std;


int solution(int left, int right) {
    int answer = 0;
    int result;
    double result2;
    for(int i=left;i<=right;i++)
    {
        result = sqrt(i);
        result2 = sqrt(i);
        if(result != result2)
        {
            answer += i;
        }
        else
        {
            answer -= i;
        }
    }
    
    return answer;
}
