#include <string>
#include <vector>
#include <algorithm>
#include <cmath>
using namespace std;

int solution(string name) {
    int answer = 0;
    string origin = "";
    for(int i=0;i<name.size();i++)
    {
        origin += 'A';
    }
    int k = name.size();
    int now = 0;
    while(k--)
    {
        if(name == origin) break;
        vector<pair<int,int>> result;
        for(int i=0;i<name.size();i++)
        {
            if(name[i] != origin[i])
            {
                int m = abs(i-now);
                int m2 = name.size();
                int dis = min(m,m2-m);
                result.push_back(make_pair(dis,i));
            }
        }
        auto it = min_element(result.begin(),result.end());
        answer += it->first;
        now = it->second;
        answer+= min(name[now]-'A',26-(name[now]-'A'));
        origin[now] = name[now];
    }
    return answer;
}

#include <string>

#include <algorithm>
#include <cmath>
using namespace std;

int solution(string name) {
    int answer = 0;
    vector<int> indexs;
    for(int i=0;i<name.size();i++)
    {
        if(name[i] != 'A') indexs.push_back(i);
    }
    int start = 0;
    while(indexs.size()!=0)
    {
        vector<pair<int,int>> moves;
        int mini = -1;
        int end = 0;
        int k = 0;
        for(int i=0;i<indexs.size();i++)
        {
            int a = abs(indexs[i]-start);
            int b = name.size()-indexs[i]+start;
            int temp = min(a,b);
            if(mini==-1 || mini > temp)
            {
                mini = temp;
                end = indexs[i];
                k = i;
            }
        }
        answer += min(name[end]-'A',26-name[end]+'A');
        answer += mini;
        start = end;
        indexs.erase(indexs.begin()+k);
    }
    return answer;
}
