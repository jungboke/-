#include <string>
#include <vector>

using namespace std;

string solution(string number, int k) {
    string answer = "";
    int size = number.size() - k;
    int first = 0;
    while(answer.size() != number.size()-k)
    {
        int max = -1;
        for(int i=first;i<=number.size()-size;i++)
        {
            if((number[i]-48)>max)
            {
                max = (number[i]-48);
                first = i+1;
            }
        }
        answer += to_string(max);
        size--;
    }
    return answer;
}

#include <string>

using namespace std;

string solution(string number, int k) {
    string answer = "";
    int ans_size = number.size() - k;
    int start = 0;
    while(answer.size()!=ans_size)
    {
        int maxi = -1;
        int idx = 0;
        for(int i=start;i<=number.size()-ans_size+answer.size();i++)
        {
            if(number[i]-48>maxi)
            {
                maxi = number[i]-48;
                idx = i;
            }
        }
        answer += to_string(maxi);
        start = idx+1;
    }
    return answer;
}
