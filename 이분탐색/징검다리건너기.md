#include <string>
#include <vector>

using namespace std;

int solution(vector<int> stones, int k) {
    int answer = 0;
    while(true)
    {
        int count = 0;
        int flag = true;
        int flag2 = false;
        for(int i=0;i<stones.size();i++)
        {
            if(stones[i] > 0)
            {
                stones[i]--;
                flag2 = false;
            }
            else 
            {
                if(flag2 == true) count++;
                else
                {
                    count=1;
                    flag2 = true;
                }
            }
            if(count == k)
            {
                flag = false;
                break;
            }
        }
        if(flag == true) answer++;
        if(flag == false) break;
    }
    return answer;
}

{
    bool flag = true;
    for(int i=0;i<stones.size();i++)
    {
        stones[i] = stones[i] - (mid-1);
    }
    int cnt = 0;
    for(int i=0;i<stones.size();i++)
    {
        if(stones[i] <= 0)
        {
            cnt++;
            if(cnt==k)
            {
                flag = false;
                break;
            }
        }
        else cnt = 0;
    }
    return flag;
}
int solution(vector<int> stones, int k) {
    int answer = 0;
    long long start = 1;
    long long end = *max_element(stones.begin(),stones.end());
    long long maxi = 0;
    while(start<=end)
    {
        long long mid = (start + end) / 2;
        if(check(stones,k,mid)==true)
        {
            start = mid + 1;
            if(maxi < mid) maxi = mid;
        }
        else
        {
            end = mid - 1;
        }
    }
    answer = maxi;
    return answer;
}
